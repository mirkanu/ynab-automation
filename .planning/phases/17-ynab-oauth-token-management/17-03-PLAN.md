  <action>
    Create src/app/api/ynab/authorize/route.ts:
    ```typescript
    import { auth } from '@/lib/auth'
    import { NextResponse } from 'next/server'

    export async function GET() {
      const session = await auth()
      if (!session?.user?.id) {
        return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
      }

      const authUrl = new URL('https://app.ynab.com/oauth/authorize')
      authUrl.searchParams.set('client_id', process.env.YNAB_CLIENT_ID!)
      authUrl.searchParams.set('redirect_uri', `${process.env.APP_URL}/api/ynab/callback`)
      authUrl.searchParams.set('response_type', 'code')

      return NextResponse.redirect(authUrl)
    }
    ```

    NOTE: YNAB's OAuth does not require PKCE for confidential clients (server-side). The `state` param is desirable for CSRF protection but YNAB does not enforce it — omit for now and add in a future hardening pass if needed (keeping scope small for v5.0 beta).

    Create src/app/api/ynab/status/route.ts:
    - auth() check → 401 if no session
    - prisma.user.findUnique select oauthToken
    - return { connected: !!user?.oauthToken }
  </action>

    Create src/app/api/ynab/callback/route.ts following the Pattern 1 example from RESEARCH.md:

    - Auth check: return 401 if no session.user.id

    - POST to https://app.ynab.com/oauth/token with:
      { code, client_id: YNAB_CLIENT_ID, client_secret: YNAB_CLIENT_SECRET, grant_type: 'authorization_code', redirect_uri: APP_URL/api/ynab/callback }
    - If tokenRes.ok is false: log status only (NOT the response body which may contain the code), return 500
