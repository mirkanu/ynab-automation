Each step page is a `'use client'` component. Common pattern for all steps:

```
'use client'
import { useState } from 'react'
import { useRouter } from 'next/navigation'

const S = { /* same inline-style constants */ }

export default function SetupStepN() {
  const router = useRouter()
  const [value, setValue] = useState('')
  const [saving, setSaving] = useState(false)
  const [error, setError] = useState('')

  async function handleNext() {
    setSaving(true)
    setError('')
    const res = await fetch('/api/setup/step', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ step: N, settings: { KEY: value } }),
    })
    const data = await res.json()
    if (!res.ok || data.error) {
      setError(data.error ?? 'Failed to save. Please try again.')
      setSaving(false)
      return
    }
    router.push(data.nextStep >= 7 ? '/setup/done' : `/setup/${data.nextStep}`)
  }
  // ... render
}
```
