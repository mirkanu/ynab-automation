# Plan 04-01 — Create Resend notification lib

## Goal

Install the Resend SDK and create `src/lib/notify.ts` — a fire-and-forget email notification helper used by the webhook handler.

## Tasks

1. Install `resend` npm package
2. Create `src/lib/notify.ts` with a `sendErrorNotification` function
3. Set `RESEND_API_KEY` and `MANUEL_EMAIL` environment variables on Railway via Railway CLI

## Implementation

### `src/lib/notify.ts`

```typescript
import { Resend } from 'resend';

const MANUEL_EMAIL = process.env.MANUEL_EMAIL ?? 'manuelkuhs@gmail.com';

export interface NotificationOptions {
  to: string;          // primary recipient email
  cc?: string;         // optional CC (used when Emily-Kate fails → CC Manuel)
  subject: string;
  body: string;        // plain-text body
}

/**
 * Sends an error notification email via Resend.
 * Fire-and-forget — never throws. Logs on failure.
 */
export async function sendErrorNotification(opts: NotificationOptions): Promise<void> {
  try {
    const resend = new Resend(process.env.RESEND_API_KEY);
    await resend.emails.send({
      from: 'YNAB Automation <notifications@resend.dev>',
      to: opts.to,
      cc: opts.cc,
      subject: opts.subject,
      text: opts.body,
    });
  } catch (err) {
    console.error('sendErrorNotification failed:', err);
  }
}

export { MANUEL_EMAIL };
```
